	removeZerosFromColumn: async function(y, direction){
		var size = this.__matrix.length;
		
		var promises = [];
	
		var column = utils.extractColumn(this.__matrix, y);
		
		var lastPosition = -1;
		var searchIndex = -1;
	
		while((searchIndex = (direction==1 ? this.getLastNotNullPosition(column) : this.getFirstNotNullPosition(column))) > -1)
		{
			if(searchIndex!=lastPosition)
			{
				for(var i=searchIndex; (direction==1 ? i < size-1 : i > 0); i = i + direction)
				{
					if(column[i+direction].getValue()==0)
					{		
						column[i + direction] = this.__matrix[i][y];
						column[i] = new Tile({}, 0);		
						
						promises.push(ui.swapTiles(i + direction, y, i, y));
					}
					else
					{
						break;
					}
				}

				lastPosition = searchIndex;
			} 
			else
			{
				lastPosition = -1;
				break;
			}
			
		}
		
		await Promise.all(promises);
	},
